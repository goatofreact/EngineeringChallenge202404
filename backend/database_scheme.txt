CREATE TABLE users (
    ID int NOT NULL AUTO_INCREMENT,
    username varchar(255),
    password varchar(255),
    session varchar(255) DEFAULT (uuid()),
    lastLogin  TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (ID)
);

DROP TABLE  weldingRobot;
DROP TABLE  paintingStation;
DROP TABLE  assemblyLine;
DROP TABLE  qualityControlStation;

CREATE TABLE weldingRobot (
    ID int NOT NULL AUTO_INCREMENT,
    errorRate float,
vibrationLevel float,
electrodeWear float,
shieldingPressure float,
wireFeedRate float,
arcStability float,
seamWidth float,
coolingEfficiency float,
    UNIQUE (ID)
);

CREATE TABLE paintingStation (
   ID int NOT NULL AUTO_INCREMENT,
    flowRate float,
pressure float,
colorConsistency float,
nozzleCondition float,
    UNIQUE (ID)
);

CREATE TABLE assemblyLine (
   ID int NOT NULL AUTO_INCREMENT,
    alignmentAccuracy float,
speed float,
fittingTolerance float,
beltSpeed float,
    UNIQUE (ID)
);


CREATE TABLE qualityControlStation (
    ID int NOT NULL AUTO_INCREMENT,
    cameraCalibration float,
lightIntensity float,
softwareVersion float,
criteriaSettings float,
    UNIQUE (ID)
);
